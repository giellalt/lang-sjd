! Personal pronouns
! =================

! a = stem му-, то̄-, со̄-
! b = stem мы-, то̄-, со̄-
! c = stem мы-, ты-, сы-
! d = stem мӣ-, тӣ -, сӣ -

LEXICON Personal

мунн:мы  pers_b ;
мунн:мы  pers_c ;
мунн:мӣ  pers_d ;

то̄нн:то̄  pers_b ;
